# 创建虚拟节点

h(当前要渲染的组件/标签字符串/函数,配置对象：{}, 内容：字符串/数组)
1
h(CountTo, {
  <!-- 设置html属性 -->
  attrs: {
      id: 'id'
  },
  <!-- 设置样式 -->
  style: {
      color: 'red'
  },
  <!-- 组件传值 -->
  props: {
    endVal: 100
  },
  <!-- DOM上的属性 -->
  domProps: {
      innerHTML: ''
  },
  <!-- 事件调用 -->
  on: {
    'on-animation-end': (val) => {
      console.log('animation end!')
    }
  },
  <!-- 给组件绑定事件 -->
  nativeOn: {
    'click': () => {
      console.log('click!')
    }
  },
  <!-- 定义class的三种方式 -->
  'class': 'count-up',
  'class': ['count-up', true ? 'aa' : 'bb'],
  'class': {
    'count-up': true
  },
  <!-- 添加指令 -->
  directives: [],
  <!-- 插槽 -->
  scopedSlots: {},
  slot: '',
  <!-- 给组件生成独立的key -->
  key: '',
  <!-- 获取html DOM或者组件实例 -->
  ref: ''
}, 'Ella')
